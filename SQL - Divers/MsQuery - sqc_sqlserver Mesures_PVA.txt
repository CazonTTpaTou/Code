SELECT VUE_METIER_MESURES.Société, VUE_METIER_MESURES.Produit, VUE_METIER_MESURES.Equipement, VUE_METIER_MESURES."Carte de contrôle", VUE_METIER_MESURES."Mesure Type", VUE_METIER_MESURES."Mesure Date Heure", VUE_METIER_MESURES."Mesure Date", DATEDIFF(day, VUE_METIER_MESURES."Mesure Date", GETDATE()) AS "Historique jour", VUE_METIER_MESURES."Mesure Jour", VUE_METIER_MESURES."Mesure Heure", VUE_METIER_MESURES."Mesure Opérateur", VUE_METIER_MESURES."Mesure Equipe", VUE_METIER_MESURES."Mesure Poste", VUE_METIER_MESURES."Mesure Date Poste",  VUE_METIER_MESURES."Mesure Observation", VUE_METIER_MESURES."Valeur Minimale Dérogation", VUE_METIER_MESURES."Valeur Minimale Critique", VUE_METIER_MESURES."Valeur mesurée", VUE_METIER_MESURES."Valeur Maximale Critique", VUE_METIER_MESURES."Valeur Maximale Dérogation", VUE_METIER_MESURES."Dérogation Code", VUE_METIER_MESURES."Dérogation Date Début", VUE_METIER_MESURES."Dérogation Date Fin", VUE_METIER_MESURES."Dérogation Commentaire", GETDATE() AS 'Date heure exe'
FROM SQC_DB.dbo.VUE_METIER_MESURES VUE_METIER_MESURES
WHERE (DATEDIFF(month,"Mesure Date",GETDATE())<=3)
AND VUE_METIER_MESURES.Equipement = 'Sérigraphie PVA'
ORDER BY VUE_METIER_MESURES."Mesure Date" DESC
