Client Requête

SELECT Client.[Code Client], Client.Nom, Commande.[Taux Remise], [Ligne Commande].Quantité, Produit.[Prix Produit], (1-[Taux Remise])*[Quantité]*[Prix Produit] AS Panier
FROM Client INNER JOIN ((Commande INNER JOIN [Ligne Commande] ON Commande.[N° Commande] = [Ligne Commande].[N° Commande]) INNER JOIN Produit ON [Ligne Commande].[Code Produit] = Produit.[Code Produit]) ON Client.[Code Client] = Commande.[N° Client];

Client Requête 2

FROM [Client Requête]
GROUP BY [Client Requête].[Code Client], [Client Requête].Nom;

Client Requête 3

SELECT [Client Requête 2].[Code Client], [Client Requête 2].Nom, [Client Requête 2].SommeDePanier AS Total
FROM [Client Requête 2]

Commande Requête 

SELECT Commande.[N° Commande], Commande.Date, [Ligne Commande].Quantité, Produit.[Prix Produit], Produit.[Taux TVA], Commande.[Taux Remise], Sum([Prix Produit]*(1+[Taux TVA])*[Quantité]*(1-[Taux Remise])) AS Total, Client.Nom, Client.Prénom, Client.[Code Postal], Left([Code Postal],2) AS CP
FROM ((Commande INNER JOIN [Ligne Commande] ON Commande.[N° Commande] = [Ligne Commande].[N° Commande]) INNER JOIN Produit ON [Ligne Commande].[Code Produit] = Produit.[Code Produit]) INNER JOIN Client ON Commande.[N° Client] = Client.[Code Client]
GROUP BY Commande.[N° Commande], Commande.Date, [Ligne Commande].Quantité, Produit.[Prix Produit], Produit.[Taux TVA], Commande.[Taux Remise], Client.Nom, Client.Prénom, Client.[Code Postal], Left([Code Postal],2);

FROM [Commande Requête]
GROUP BY [Commande Requête].[N° Commande], [Commande Requête].Date, [Commande Requête].Nom, [Commande Requête].Prénom, [Commande Requête].CP;
