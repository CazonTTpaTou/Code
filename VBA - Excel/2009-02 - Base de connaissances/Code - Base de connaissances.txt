Public Function PCase(AnyText As String) As String

Dim FixedText As String

FixedText = StrConv(AnyText, 3)
If Left(FixedText, 2) = "Mc" Then FixedText = Left(FixedText, 2) + UCase(Mid(FixedText, 3, 1)) + Mid(FixedText, 4)
End If
PCase = FixedText

End Function
