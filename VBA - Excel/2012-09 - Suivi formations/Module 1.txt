Sub insertDate(dateInser As Date)

Dim annee(1 To 14) As Integer

annee(1) = 2009
annee(2) = 2010
annee(3) = 2011
annee(4) = 2012
annee(5) = 2013
annee(6) = 2014
annee(7) = 2015
annee(8) = 2016
annee(9) = 2017
annee(10) = 2018
annee(11) = 2019
annee(12) = 2020

ligne = ActiveCell.Row
Col = ActiveCell.Column

DateInsert = Format(dateInser, "mmm")
ActiveCell.value = DateInsert
Cells(ActiveCell.Row, 25).value = DateAdd("m", delai_base_BPL, dateInser)
Cells(ActiveCell.Row, 26).value = DateAdd("m", delai_urgent_BPL, dateInser)

For i = 2 To 13
    Cells(ligne, i).Select
    Selection.ClearComments
    If annee(i - 1) = Year(DateAdd("m", delai_base_BPL, dateInser)) Then
        'MsgBox ("valeur de i " & i & "Année " & annee(i)) & " Year : " & Year(DateAdd("m", delai_base_BPL, dateInser))
        Message = "Formation " & Cells(5, 25).value & Chr(10) & "Pour " & Cells(ligne, 1).value & Chr(10) & "Prévue entre : " & Chr(10)
        Message = Message & "Le " & DateAdd("m", delai_base_BPL, dateInser) & Chr(10) & "Et le " & DateAdd("m", delai_urgent_BPL, dateInser)
        Selection.AddComment
        Selection.Comment.Visible = False
        Selection.Comment.text text:=Message
    End If
Next i

Cells(ligne, Col + 2).Select

End Sub

Sub Calendrier(ByVal Target As Range)

Set ZoneD = Range("B6:M1000")

' On détermine l'heure et la date courante
annee = Target.Column - 2
moisM = Month(Date) - 1
jour = 0

' Si la cellule cliquée appartient à la zone de date détectée alors on affiche le formulaire de date
If Not ((Intersect(Target, ZoneD) Is Nothing)) Then

    UserForm1.ComboBox1.ListIndex = moisM
    UserForm1.ComboBox2.ListIndex = jour
    UserForm1.ComboBox3.ListIndex = annee
    UserForm1.ComboBox3.Enabled = False
    UserForm1.ComboBox2.Enabled = False
    UserForm1.Show
   
End If

End Sub
