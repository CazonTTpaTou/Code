using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace WebApplication1
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        private Button b;
        string nom_produit = "Livres";
        int id_produit = 34;

        protected override void OnInit(EventArgs e)
        {
            base.OnInit(e);
            Calendar1.DayRender += new DayRenderEventHandler(Calendar1_DayRender);
        }

        void Calendar1_DayRender(object sender, DayRenderEventArgs e)
        {
            //b.Text = "caiman";
        }

        protected void Page_Load(object sender, EventArgs e)
        {

            if(! IsPostBack)
            {
                ListBox1.Items.Add("Nimes");
                ListBox1.Items.Add("Hyères");
                ListBox1.Items.Add("Porquerolles");
            }

            // Création dynamique du bouton
            b = new Button();
            b.Text = "Cliquez ici";
            b.ID = "bouton";

            // Le bouton doit être un descendant du formulaire
            form1.Controls.Add(b);

            // inscription d'un gestionnaire d'évènements
            b.Click += new EventHandler(b_click);

            // Définition du lien hyper texte
            HyperLink1.Text = "Produit" + nom_produit;
            HyperLink1.NavigateUrl = string.Format("WebForm1.aspx?id_produit={0}",id_produit);

        }

        protected void Calendar1_SelectionChanged(object sender,EventArgs e)
        {
            b.Text = "Felin";
        }
        void b_click(object sender,EventArgs e)
        {
            Label1.Text = "Vous avez déjà cliqué !!";
            Response.Redirect("Default.aspx");
        }
    }
}
